# Installation Instructions
# ============================================================================

# To install all dependencies:
#   pip install -r requirements.txt

# To install with specific Spark/Java versions:
#   1. Install Java 11: brew install openjdk@11 (macOS)
#   2. Set JAVA_HOME: export JAVA_HOME=$(/usr/libexec/java_home -v 11)
#   3. Install requirements: pip install -r requirements.txt

# For GraphFrames, you also need to configure Spark:
#   spark = SparkSession.builder \
#       .config("spark.jars.packages", "graphframes:graphframes:0.8.2-spark3.2-s_2.12") \
#       .getOrCreate()

# For BigQuery connector:
#   spark = SparkSession.builder \
#       .config("spark.jars.packages", "com.google.cloud.spark:spark-bigquery-with-dependencies_2.12:0.27.1") \
#       .getOrCreate()
